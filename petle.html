8.2 Pętla for

for (element in wektor) {
  przetwarzanie elementu
}
odl_mile = c(142, 63, 121)
for (i in odl_mile) {
  print(i * 1.609)
}
#> [1] 228
#> [1] 101
#> [1] 195
for (i in 1:3) {
  print(odl_mile[i] * 1.609)
}
#> [1] 228
#> [1] 101
#> [1] 195
odl_mile_l = seq_along(odl_mile)
for (i in odl_mile_l) {
  print(odl_mile[i] * 1.609)
}
#> [1] 228
#> [1] 101
#> [1] 195
odl_mile_l = seq_along(odl_mile)
for (i in odl_mile_l) {
  odl_mile[i] = odl_mile[i] * 1.609
}
odl_mile
#> [1] 228 101 195
odl_km = vector("numeric", length = 0)
odl_mile_l = seq_along(odl_mile)
for (i in odl_mile_l) {
  odl_km = c(odl_km, odl_mile[i] * 1.609)
}
odl_km
#> [1] 368 163 313
odl_mile = c(142, 63, 121)
odl_km = vector("numeric", length = length(odl_mile))
odl_mile_l = seq_along(odl_mile)
for (i in odl_mile_l) {
  odl_km[i] = odl_mile[i] * 1.609
}
odl_km
#> [1] 228 101 195

for (i in c(1, 2, 4, 5, 6)){

      print("Działa!")
}
  1. Spójrz na poniższy kod, ale nie wykonuj go. Ile razy zostanie wyświetlony tekst "Działa!"?
for (i in c(1, 2, 4, 5, 6)){
  for (j in 6:3){
    if (i < 2 | i >= 5)
      print("Działa!")
  }
}
  1. Spójrz na poniższy kod, ale nie wykonuj go. Ile razy zostanie wyświetlony tekst "Działa!"?
for (i in c(1, 2, 4, 5, 6)){
  for (j in 6:3){
    if (i < 2 & j >= 5)
      print("Działa!")
  }
}
  1. Spójrz na poniższy kod, ale nie wykonuj go. Ile razy zostanie wyświetlony tekst "Działa!"?
for (i in c(1, 2, 4, 5, 6)){
  for (j in 6:3){
    if (i < 2 | j >= 5)
      print("Działa!")
  }
}
  1. Spójrz na poniższy kod, ale nie wykonuj go. Ile razy zostanie wyświetlony tekst "Działa!"?
for (i in c(1, 2, 3)){
  for (j in 6:3){
    i = i + j
    if (i < 4 | i >= 9)
      print("Działa!")
  }
}
